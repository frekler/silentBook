```js
let arr = [
    { name: 'Вася', age: 36 },
    { name: 'Коля', age: 18 },
    { name: 'Оля', age: 'не скажу' },
]

let result = arr.filter(function (item, index, array) {
    return item.age >= 18;
});
console.log(result);    // новый массив из двух элементов {name Вася..., Коля}
```

  let res = [];
